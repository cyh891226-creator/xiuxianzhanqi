# 修仙战棋 - 技术实现方案

## 一、技术选型

### 1.1 游戏引擎

#### 推荐方案：Unity 2022.3 LTS
**优势：**
- 成熟的2D/2.5D战棋游戏开发方案
- 丰富的Asset Store资源
- 跨平台支持（PC、移动端、主机）
- 完善的文档和社区支持
- C#开发，团队易上手

**备选方案：**
- **Godot 4.x**：开源免费，轻量级，GDScript/Python语法
- **Cocos Creator**：国产引擎，对中文支持好，2D游戏优秀
- **Unreal Engine 5**：3D效果出色，但2D战棋有点大材小用

### 1.2 开发语言
- **主要语言**：C# (Unity)
- **配置文件**：JSON / ScriptableObject
- **数据表格**：Excel → CSV / JSON 转换

### 1.3 版本控制
- **Git**：代码版本管理
- **Git LFS**：大文件管理（美术资源、音频等）

```csharp
// 事件中心 - 解耦模块间通信
public class EventCenter
{
    private static Dictionary<string, Action<object>> eventDict = new Dictionary<string, Action<object>>();
    
    // 添加监听
    public static void AddListener(string eventName, Action<object> callback)
    {
        if (!eventDict.ContainsKey(eventName))
            eventDict[eventName] = callback;
        else
            eventDict[eventName] += callback;
    }
    
    // 移除监听
    public static void RemoveListener(string eventName, Action<object> callback)
    {
        if (eventDict.ContainsKey(eventName))
            eventDict[eventName] -= callback;
    }
    
    // 触发事件
    public static void TriggerEvent(string eventName, object param = null)
    {
        if (eventDict.ContainsKey(eventName))
            eventDict[eventName]?.Invoke(param);
    }
}

// 使用示例
// 订阅事件
EventCenter.AddListener("OnBattleStart", OnBattleStart);

// 触发事件
```

```csharp
// 单位基类
public abstract class UnitBase : MonoBehaviour
{
    [Header("基础信息")]
    public string unitName;
    public UnitType unitType;
    public Faction faction;
    
    [Header("战斗属性")]
    public CharacterStats baseStats;      // 基础属性
    public CharacterStats currentStats;   // 当前属性
    
    [Header("状态")]
    public Vector2Int gridPosition;
    public bool isAlive = true;
    public bool hasMoved = false;
    public bool hasActed = false;
    
    [Header("资源")]
    public int currentHP;
    public int currentMP;
    public int currentRage;
    
    // 组件引用
    protected UnitView unitView;
    protected UnitAI unitAI;
    
    // 事件
    public event Action OnUnitSelected;
    public event Action OnUnitMoved;
    public event Action OnUnitActed;
    public event Action OnUnitDeath;
    
    // 初始化
    public virtual void Initialize(CharacterData data)
    {
        unitName = data.characterName;
        baseStats = data.baseStats;
        currentStats = baseStats;
        currentHP = currentStats.maxHP;
        currentMP = currentStats.maxMP;
    }
    
    // 移动到指定格子
    public virtual void MoveTo(Vector2Int targetPos)
    {
        // 播放移动动画
        StartCoroutine(MoveCoroutine(targetPos));
    }
    
    protected IEnumerator MoveCoroutine(Vector2Int targetPos)
    {
        Vector3 targetWorldPos = GridSystem.Instance.GridToWorld(targetPos);
        
        while (Vector3.Distance(transform.position, targetWorldPos) > 0.01f)
        {
            transform.position = Vector3.MoveTowards(transform.position, targetWorldPos, 5f * Time.deltaTime);
            yield return null;
        }
        
        gridPosition = targetPos;
        hasMoved = true;
        OnUnitMoved?.Invoke();
    }
    
    // 使用技能
    public virtual void UseSkill(SkillBase skill, List<UnitBase> targets)
    {
        if (currentMP < skill.mpCost) return;
        
        currentMP -= skill.mpCost;
        skill.Execute(this, targets);
        hasActed = true;
        OnUnitActed?.Invoke();
    }
    
    // 受到伤害
    public virtual void TakeDamage(int damage, DamageType damageType)
    {
        // 计算实际伤害
        int actualDamage = CalculateDamage(damage, damageType);
        currentHP -= actualDamage;
        
        // 播放受击动画
        unitView.PlayHitAnimation();
        
        // 显示伤害数字
        DamageNumberManager.Instance.ShowDamage(actualDamage, transform.position);
        
        if (currentHP <= 0)
        {
            Die();
        }
    }
    
    // 计算伤害
    protected virtual int CalculateDamage(int rawDamage, DamageType damageType)
    {
        float defense = damageType == DamageType.Physical ? currentStats.defense : currentStats.magicDefense;
        float reduction = defense / (defense + 100f); // 防御公式
        return Mathf.Max(1, Mathf.RoundToInt(rawDamage * (1 - reduction)));
    }
    
    // 死亡
    protected virtual void Die()
    {
        isAlive = false;
        unitView.PlayDeathAnimation();
        OnUnitDeath?.Invoke();
    }
}
```

```csharp
// 技能基类
public abstract class SkillBase : MonoBehaviour
{
    public SkillData skillData;
    protected int currentCooldown;
    
    // 检查是否可以使用
    public virtual bool CanUse(UnitBase caster)
    {
        return caster.currentMP >= skillData.mpCost 
            && caster.currentRage >= skillData.rageCost
            && currentCooldown <= 0;
    }
    
    // 执行技能
    public virtual void Execute(UnitBase caster, List<UnitBase> targets)
    {
        // 播放施法动画
        StartCoroutine(CastSkill(caster, targets));
    }
    
    protected IEnumerator CastSkill(UnitBase caster, List<UnitBase> targets)
    {
        // 施法前摇
        yield return new WaitForSeconds(0.3f);
        
        // 播放特效
        if (skillData.castEffect != null)
        {
            Instantiate(skillData.castEffect, caster.transform.position, Quaternion.identity);
        }
        
        // 应用效果
        foreach (var target in targets)
        {
            ApplyEffects(caster, target);
        }
        
        // 进入冷却
        currentCooldown = skillData.cooldown;
    }
    
    // 应用效果
    protected virtual void ApplyEffects(UnitBase caster, UnitBase target)
    {
        foreach (var effect in skillData.effects)
        {
            effect.Apply(caster, target);
        }
    }
    
    // 冷却回合减少
    public void ReduceCooldown()
    {
        if (currentCooldown > 0)
            currentCooldown--;
    }
}

// 伤害技能
public class DamageSkill : SkillBase
{
    protected override void ApplyEffects(UnitBase caster, UnitBase target)
    {
        base.ApplyEffects(caster, target);
        
        // 计算伤害
        int damage = CalculateDamage(caster, target);
        target.TakeDamage(damage, DamageType.Physical);
    }
    
    int CalculateDamage(UnitBase caster, UnitBase target)
    {
        float attack = caster.currentStats.attack;
        float multiplier = 1.5f; // 技能系数
        return Mathf.RoundToInt(attack * multiplier);
    }
}

// 治疗技能
public class HealSkill : SkillBase
{
    protected override void ApplyEffects(UnitBase caster, UnitBase target)
    {
        base.ApplyEffects(caster, target);
        
        int healAmount = CalculateHeal(caster, target);
        target.currentHP = Mathf.Min(target.currentHP + healAmount, target.currentStats.maxHP);
    }
    
    int CalculateHeal(UnitBase caster, UnitBase target)
    {
        float magicAttack = caster.currentStats.magicAttack;
        return Mathf.RoundToInt(magicAttack * 2f);
    }
}
```

```csharp
// AI决策
public class UnitAI : MonoBehaviour
{
    private UnitBase unit;
    private AIBehavior behavior;
    
    void Awake()
    {
        unit = GetComponent<UnitBase>();
    }
    
    // AI回合决策
    public IEnumerator MakeDecision()
    {
        // 1. 评估局势
        var situation = EvaluateSituation();
        
        // 2. 选择行为
        behavior = SelectBehavior(situation);
        
        // 3. 执行行为
        yield return StartCoroutine(ExecuteBehavior(behavior));
    }
    
    // 评估局势
    AI_Situation EvaluateSituation()
    {
        var situation = new AI_Situation();
        
        // 获取所有单位
        var allUnits = BattleManager.Instance.GetAllUnits();
        
        // 评估威胁
        situation.threats = allUnits
            .Where(u => u.faction != unit.faction && u.isAlive)
            .OrderBy(u => Vector2Int.Distance(u.gridPosition, unit.gridPosition))
            .ToList();
        
        // 评估友军
        situation.allies = allUnits
            .Where(u => u.faction == unit.faction && u.isAlive && u != unit)
            .ToList();
        
        // 评估自身状态
        situation.selfHealthPercent = (float)unit.currentHP / unit.currentStats.maxHP;
        
        return situation;
    }
    
    // 选择行为
    AIBehavior SelectBehavior(AI_Situation situation)
    {
        // 简单AI：优先攻击最近的敌人
        if (situation.threats.Count > 0)
        {
            var target = situation.threats[0];
            
            // 检查是否在攻击范围内
            int distance = Vector2Int.Distance(unit.gridPosition, target.gridPosition);
            
            if (distance <= unit.currentStats.attackRange)
            {
                // 直接攻击
                return new AIBehavior { type = AIActionType.Attack, target = target };
            }
            else
            {
                // 移动到攻击范围
                return new AIBehavior { type = AIActionType.Move, target = target };
            }
        }
        
        return new AIBehavior { type = AIActionType.Wait };
    }
    
    // 执行行为
    IEnumerator ExecuteBehavior(AIBehavior behavior)
    {
        switch (behavior.type)
        {
            case AIActionType.Move:
                yield return StartCoroutine(MoveTowardsTarget(behavior.target));
                break;
            case AIActionType.Attack:
                yield return StartCoroutine(AttackTarget(behavior.target));
                break;
            case AIActionType.Wait:
                yield return new WaitForSeconds(0.5f);
                break;
        }
    }
    
    // 向目标移动
    IEnumerator MoveTowardsTarget(UnitBase target)
    {
        // 计算移动路径
        var path = GridSystem.Instance.FindPath(unit.gridPosition, target.gridPosition, unit.currentStats.moveRange);
        
        // 移动到最远可达位置
        if (path.Count > 0)
        {
            Vector2Int movePos = path[Mathf.Min(path.Count - 1, unit.currentStats.moveRange)];
            unit.MoveTo(movePos);
            yield return new WaitUntil(() => !unit.hasMoved);
        }
    }
    
    // 攻击目标
    IEnumerator AttackTarget(UnitBase target)
    {
        // 使用普通攻击或技能
        unit.UseSkill(unit.GetComponent<SkillBase>(), new List<UnitBase> { target });
        yield return new WaitForSeconds(1f);
    }
}
```

```csharp
// 存档管理器
public class SaveSystem : Singleton<SaveSystem>
{
    private string SavePath => Application.persistentDataPath + "/Saves/";
    private const string SAVE_EXTENSION = ".sav";
    
    void Awake()
    {
        // 确保存档目录存在
        if (!Directory.Exists(SavePath))
            Directory.CreateDirectory(SavePath);
    }
    
    // 保存游戏
    public void SaveGame(string saveName)
    {
        SaveData saveData = new SaveData
        {
            saveName = saveName,
            saveTime = DateTime.Now.ToString("yyyy-MM-dd HH:mm:ss"),
            playTime = GameManager.Instance.TotalPlayTime,
            playerData = GameManager.Instance.PlayerData,
            teamMembers = GetTeamMembersData(),
            currentChapter = GameManager.Instance.CurrentChapter,
            currentLevel = GameManager.Instance.CurrentLevel,
            completedLevels = GameManager.Instance.CompletedLevels,
            inventory = GetInventoryData(),
            settings = SettingsManager.Instance.CurrentSettings
        };
        
        string json = JsonUtility.ToJson(saveData, true);
        string filePath = SavePath + saveName + SAVE_EXTENSION;
        
        File.WriteAllText(filePath, json);
        
        Debug.Log($"游戏已保存: {filePath}");
    }
    
    // 加载游戏
    public SaveData LoadGame(string saveName)
    {
        string filePath = SavePath + saveName + SAVE_EXTENSION;
        
        if (File.Exists(filePath))
        {
            string json = File.ReadAllText(filePath);
            SaveData saveData = JsonUtility.FromJson<SaveData>(json);
            return saveData;
        }
        
        Debug.LogError($"存档不存在: {filePath}");
        return null;
    }
    
    // 获取所有存档
    public List<SaveData> GetAllSaves()
    {
        List<SaveData> saves = new List<SaveData>();
        
        if (Directory.Exists(SavePath))
        {
            string[] files = Directory.GetFiles(SavePath, "*" + SAVE_EXTENSION);
            
            foreach (string file in files)
            {
                string json = File.ReadAllText(file);
                SaveData saveData = JsonUtility.FromJson<SaveData>(json);
                saves.Add(saveData);
            }
        }
        
        return saves.OrderByDescending(s => s.saveTime).ToList();
    }
    
    // 删除存档
    public void DeleteSave(string saveName)
    {
        string filePath = SavePath + saveName + SAVE_EXTENSION;
        
        if (File.Exists(filePath))
        {
            File.Delete(filePath);
            Debug.Log($"存档已删除: {filePath}");
        }
    }
    
    // 自动存档
    public void AutoSave()
    {
        SaveGame("AutoSave");
    }
}
```

```csharp
// Excel读取工具
public class ExcelToJsonConverter
{
    public static void ConvertExcelToJson(string excelPath, string outputPath)
    {
        // 使用EPPlus或NPOI读取Excel
        using (var package = new ExcelPackage(new FileInfo(excelPath)))
        {
            var worksheet = package.Workbook.Worksheets[0];
            
            // 读取表头
            var headers = new List<string>();
            for (int col = 1; col <= worksheet.Dimension.Columns; col++)
            {
                headers.Add(worksheet.Cells[1, col].Text);
            }
            
            // 读取数据
            var dataList = new List<Dictionary<string, object>>();
            
            for (int row = 2; row <= worksheet.Dimension.Rows; row++)
            {
                var rowData = new Dictionary<string, object>();
                
                for (int col = 1; col <= worksheet.Dimension.Columns; col++)
                {
                    string header = headers[col - 1];
                    string value = worksheet.Cells[row, col].Text;
                    rowData[header] = ParseValue(value);
                }
                
                dataList.Add(rowData);
            }
            
            // 转换为JSON
            string json = JsonConvert.SerializeObject(dataList, Formatting.Indented);
            File.WriteAllText(outputPath, json);
        }
    }
    
    static object ParseValue(string value)
    {
        // 尝试解析为整数
        if (int.TryParse(value, out int intValue))
            return intValue;
        
        // 尝试解析为浮点数
        if (float.TryParse(value, out float floatValue))
            return floatValue;
        
        // 尝试解析为布尔值
        if (bool.TryParse(value, out bool boolValue))
            return boolValue;
        
        // 默认为字符串
        return value;
    }
}
```

```csharp
        {
            var prefab = panelPrefabs.Find(p => p.GetType() == typeof(T));

                panels[panelName] = panel;
            }
        }
        
        if (panels.ContainsKey(panelName))
        {
            var panel = panels[panelName];
            panel.Show();
            panelStack.Push(panel);
            return panel as T;
        }
        
        return null;
    }
    
    // 关闭面板
    public void ClosePanel(UIPanel panel)
    {
        panel.Hide();
        
        if (panelStack.Count > 0 && panelStack.Peek() == panel)
        {
            panelStack.Pop();
        }
    }
    
    // 返回上一级
    public void GoBack()
    {
        if (panelStack.Count > 1)
        {
            var current = panelStack.Pop();
            current.Hide();
            
            var previous = panelStack.Peek();
            previous.Show();
        }
    }
}

        StartCoroutine(FadeIn());
        IsVisible = true;
        OnShow();
    }
    
    public virtual void Hide()
    {
        StartCoroutine(FadeOut(() =>
        {
            gameObject.SetActive(false);
            IsVisible = false;
            OnHide();
        }));
    }
    
    protected virtual void OnShow() { }
    protected virtual void OnHide() { }
    
    IEnumerator FadeIn()
    {
        canvasGroup.alpha = 0;
        float timer = 0;
        
        while (timer < fadeDuration)
        {
            timer += Time.deltaTime;
            canvasGroup.alpha = timer / fadeDuration;
            yield return null;
        }
        
        canvasGroup.alpha = 1;
    }
    
    IEnumerator FadeOut(Action onComplete)
    {
        float timer = 0;
        
        while (timer < fadeDuration)
        {
            timer += Time.deltaTime;
            canvasGroup.alpha = 1 - (timer / fadeDuration);
            yield return null;
        }
        
        canvasGroup.alpha = 0;
        onComplete?.Invoke();
    }
}
```

```csharp
// 对象池管理器
public class ObjectPoolManager : Singleton<ObjectPoolManager>
{
    private Dictionary<string, Queue<GameObject>> pools = new Dictionary<string, Queue<GameObject>>();
    private Dictionary<string, GameObject> prefabs = new Dictionary<string, GameObject>();
    
    // 创建对象池
    public void CreatePool(string poolName, GameObject prefab, int initialSize = 10)
    {
        prefabs[poolName] = prefab;
        pools[poolName] = new Queue<GameObject>();
        
        for (int i = 0; i < initialSize; i++)
        {
            GameObject obj = Instantiate(prefab);
            obj.SetActive(false);
            pools[poolName].Enqueue(obj);
        }
    }
    
    // 获取对象
    public GameObject GetObject(string poolName, Vector3 position, Quaternion rotation)
    {
        if (pools.ContainsKey(poolName) && pools[poolName].Count > 0)
        {
            GameObject obj = pools[poolName].Dequeue();
            obj.transform.position = position;
            obj.transform.rotation = rotation;
            obj.SetActive(true);
            return obj;
        }
        
        // 如果池为空，创建新对象
        if (prefabs.ContainsKey(poolName))
        {
            return Instantiate(prefabs[poolName], position, rotation);
        }
        
        return null;
    }
    
    // 回收对象
    public void ReturnObject(string poolName, GameObject obj)
    {
        obj.SetActive(false);
        
        if (pools.ContainsKey(poolName))
        {
            pools[poolName].Enqueue(obj);
        }
        else
        {
            Destroy(obj);
        }
    }
}
```

```csharp
// 资源加载管理器
public class ResourceManager : Singleton<ResourceManager>
{
    private Dictionary<string, Object> cache = new Dictionary<string, Object>();
    
    // 同步加载
    public T Load<T>(string path) where T : Object
    {
        if (cache.ContainsKey(path))
        {
            return cache[path] as T;
        }
        
        T asset = Resources.Load<T>(path);
        if (asset != null)
        {
            cache[path] = asset;
        }
        
        return asset;
    }
    
    // 异步加载
    public async Task<T> LoadAsync<T>(string path) where T : Object
    {
        if (cache.ContainsKey(path))
        {
            return cache[path] as T;
        }
        
        ResourceRequest request = Resources.LoadAsync<T>(path);
        await Task.Yield();
        
        while (!request.isDone)
        {
            await Task.Yield();
        }
        
        T asset = request.asset as T;
        if (asset != null)
        {
            cache[path] = asset;
        }
        
        return asset;
    }
    
    // 释放资源
    public void Release(string path)
    {
        if (cache.ContainsKey(path))
        {
            cache.Remove(path);
        }
    }
    
    // 清空缓存
    public void ClearCache()
    {
        cache.Clear();
        Resources.UnloadUnusedAssets();
    }
}
```
